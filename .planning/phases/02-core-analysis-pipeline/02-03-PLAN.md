```typescript
export type MeaningSource = 'dictionary' | 'ai' | 'both';

export interface EnrichedWord extends AnalyzedWord {
  mw_definitions: string[];
  apte_definitions: string[];
  meaning_source: MeaningSource;
}

export interface MorphologyInfo {
  stem: string;
  word_type: 'noun' | 'verb' | 'adjective' | 'indeclinable' | 'participle' | 'pronoun';
  vibhakti?: string;
  vacana?: string;
  linga?: string;
  dhatu?: string;
  gana?: number;
  lakara?: string;
  purusha?: string;
}

export interface SamasaInfo {
  compound: string;
  is_compound: boolean;
  samasa_type?: string;
  components?: { word: string; iast: string; meaning: string; role: string }[];
}
```
